# 分布式系统设计

分布式系统是由通过网络连接的多个计算节点组成的系统，能够共同协作完成任务。

## 核心理论

### CAP 定理

```
       一致性 (C)
          |
    P ---|--- C
          |
       可用性 (A)
```

CAP 定理指出：在分布式系统中，一致性（Consistency）、可用性（Availability）、分区容错性（Partition Tolerance）这三个特性最多只能同时满足两个。

### BASE 理论

- **基本可用（Basically Available）**: 系统保证基本功能可用
- **软状态（Soft State）**: 允许数据在不同节点间存在中间状态
- **最终一致性（Eventually Consistency）**: 经过一定时间后达到一致

## 一致性模型

```python
class ConsistencyModel:
    """一致性模型"""

    def __init__(self):
        self.data = {}
        self.version = {}
        self.clock = 0

    def strong_consistency(self, key, value):
        """强一致性"""
        # 所有副本同时更新
        self.data[key] = value
        self.version[key] = self.clock + 1
        self.clock += 1
        return True

    def eventual_consistency(self, key, value):
        """最终一致性"""
        # 允许副本间有时间差
        self.data[key] = value
        self.version[key] = self.clock + 1
        self.clock += 1
        # 异步同步到其他副本
        self.propagate_changes(key, value)
        return True

    def propagate_changes(self, key, value):
        """传播变更"""
        # 模拟异步传播
        print(f"正在传播 {key} = {value} 到其他副本...")
        # 实际实现中会通过网络同步

    def check_consistency(self, key):
        """检查一致性"""
        current_version = self.version.get(key, 0)
        return current_version == self.clock
```

```python
class RaftNode:
    """Raft 节点"""

    def __init__(self, node_id, peers):
        self.node_id = node_id
        self.peers = peers
        self.state = 'follower'  # follower, candidate, leader
        self.current_term = 0
        self.voted_for = None
        self.log = []
        self.commit_index = 0
        self.last_applied = 0
        self.leader_id = None

        # Leader 特有字段
        self.next_index = {}
        self.match_index = {}

    def become_candidate(self):
        """成为候选人"""
        self.state = 'candidate'
        self.current_term += 1
        self.voted_for = self.node_id
        self.request_votes()

    def request_votes(self):
        """请求投票"""
        votes = 1  # 自己投给自己

        for peer in self.peers:
            if self.request_vote(peer):
                votes += 1

        if votes > len(self.peers) // 2:
            self.become_leader()

    def become_leader(self):
        """成为领导者"""
        self.state = 'leader'
        self.leader_id = self.node_id
        print(f"节点 {self.node_id} 成为 Leader (Term {self.current_term})")

    def receive_vote(self, term, candidate_id):
        """接收投票请求"""
        if term > self.current_term:
            self.current_term = term
            self.voted_for = None
            self.state = 'follower'

        if self.voted_for is None:
            self.voted_for = candidate_id
            return True

        return False

    def append_entries(self, leader_id, term, entries):
        """接收日志追加"""
        if term < self.current_term:
            return False

        if self.state != 'follower':
            self.state = 'follower'
            self.leader_id = leader_id

        self.log.extend(entries)
        return True
```

### 两阶段提交（2PC）

```python
class TwoPhaseCommit:
    """两阶段提交"""

    def __init__(self, participants):
        self.participants = participants
        self.transaction_log = []

    def begin_transaction(self):
        """开始事务"""
        self.transaction_log.append("BEGIN TRANSACTION")
        return True

    def prepare(self):
        """准备阶段"""
        prepared = []
        for participant in self.participants:
            if participant.prepare():
                prepared.append(participant)
            else:
                # 回滚
                self.rollback()
                return False

        return len(prepared) > 0

    def commit(self):
        """提交阶段"""
        committed = []
        for participant in self.participants:
            if participant.commit():
                committed.append(participant)
            else:
                # 处理失败
                self.handle_commit_failure(participant)

        self.transaction_log.append("COMMIT")
        return True

    def rollback(self):
        """回滚"""
        for participant in self.participants:
            participant.rollback()
        self.transaction_log.append("ROLLBACK")

    def handle_commit_failure(self, participant):
        """处理提交失败"""
        # 重试或记录日志
        print(f"Participant {participant} committed failed, retrying...")
        participant.commit()

class Participant:
    """参与节点"""

    def __init__(self, name):
        self.name = name
        self.state = 'initial'
        self.transaction_log = []

    def prepare(self):
        """准备"""
        try:
            # 执行本地事务准备
            self.state = 'prepared'
            return True
        except Exception as e:
            print(f"{self.name} prepare failed: {e}")
            return False

    def commit(self):
        """提交"""
        try:
            self.state = 'committed'
            return True
        except Exception as e:
            print(f"{self.name} commit failed: {e}")
            return False

    def rollback(self):
        """回滚"""
        self.state = 'rolled_back'
```

### Saga 模式

```python
class SagaTransaction:
    """Saga 事务"""

    def __init__(self):
        self.steps = []
        self.compensating_actions = []

    def add_step(self, action, compensate):
        """添加步骤"""
        self.steps.append(action)
        self.compensating_actions.append(compensate)

    def execute(self):
        """执行 Saga 事务"""
        completed_steps = []

        for i, step in enumerate(self.steps):
            try:
                step()
                completed_steps.append(i)
            except Exception as e:
                # 执行补偿操作
                self._compensate(completed_steps)
                raise Exception(f"Saga failed at step {i}: {e}")

        return True

    def _compensate(self, completed_steps):
        """执行补偿"""
        for step_index in reversed(completed_steps):
            try:
                self.compensating_actions[step_index]()
            except Exception as e:
                print(f"Compensation failed for step {step_index}: {e}")
```

## 服务发现

```python
class ServiceRegistry:
    """服务注册中心"""

    def __init__(self):
        self.services = {}

    def register(self, service_name, instance_id, address, port):
        """注册服务"""
        if service_name not in self.services:
            self.services[service_name] = {}

        self.services[service_name][instance_id] = {
            'address': address,
            'port': port,
            'last_heartbeat': time.time()
        }
        print(f"服务 {service_name} 实例 {instance_id} 注册成功")

    def deregister(self, service_name, instance_id):
        """注销服务"""
        if service_name in self.services and instance_id in self.services[service_name]:
            del self.services[service_name][instance_id]
            print(f"服务 {service_name} 实例 {instance_id} 注销成功")

    def discover(self, service_name):
        """服务发现"""
        if service_name in self.services and self.services[service_name]:
            # 返回可用的服务实例
            available_instances = [
                instance for instance in self.services[service_name].values()
                if self._is_healthy(instance)
            ]
            return available_instances
        return None

    def _is_healthy(self, instance):
        """健康检查"""
        # 简化实现
        return time.time() - instance['last_heartbeat'] < 30

# 使用示例
registry = ServiceRegistry()
registry.register('user-service', 'instance-1', '192.168.1.1', 8001)
registry.register('user-service', 'instance-2', '192.168.1.2', 8001)

instances = registry.discover('user-service')
print(f"找到服务实例: {instances}")
```

## 最佳实践

1. **服务拆分**: 合理划分微服务边界
2. **接口设计**: 清晰的 API 定义
3. **数据一致性**: 合理选择一致性模型
4. **错误处理**: 完善的容错机制
5. **监控告警**: 全面的监控和日志
6. **测试验证**: 充分的集成测试
